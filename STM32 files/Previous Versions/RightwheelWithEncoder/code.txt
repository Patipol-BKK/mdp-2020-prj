/* USER CODE BEGIN Header_EncoderR */
/**
* @brief Function implementing the EncoderTask thread.
* @param argument: Not used
* @retval None
*/
/* USER CODE END Header_EncoderR */
void EncoderR(void *argument)
{
  /* USER CODE BEGIN EncoderR */
  /* Infinite loop */
  HAL_TIM_Encoder_Start(&htim3, TIM_CHANNEL_ALL); //start generating the timer

  int cnt1, cnt2, diff;
  uint32_t tick;
  uint8_t dir;
  uint8_t hello[20];

  cnt1 = __HAL_TIM_GET_COUNTER(&htim3);
  tick = HAL_GetTick();
  for(;;)
  {
	if(HAL_GetTick()-tick >1000L){ //every 1000 ticks
		cnt2 = __HAL_TIM_GET_COUNTER(&htim3);
		if(__HAL_TIM_IS_TIM_COUNTING_DOWN(&htim3)){
			if(cnt2 < cnt1)
				diff = cnt1 - cnt2;
			else
				diff = (65535 - cnt2) + cnt1;
		}
		else{
			if(cnt2 > cnt1)
				diff = cnt2 - cnt1;
			else
				diff = (65535 - cnt1) + cnt2;
		}
		sprintf(hello, "Speed:%5d\0", diff);
		OLED_ShowString(10,20,hello);
		dir = ____HAL_TIM_IS_TIM_COUNTING_DOWN(&htim3);
		sprintf(hello, "Dir:%5d\0", dir);
		OLED_ShowString(10,30,hello);
		cnt1 = __HAL_TIM_GET_COUNTER(&htim3);
		tick = HAL_GetTick();






	}
    osDelay(1);
  }
  /* USER CODE END EncoderR */
}
